WHERE TaskName LIKE '%æµ‹è¯•%';

-- 2. 更新乱码数据 (示例)
UPDATE task_pool_items SET TaskName = '测试任务'
WHERE TaskName = 'æµ‹è¯•ä»»åŠ¡';
